**Generated hook signatures (from lib/api-client-react/src/generated/api.ts):**
```typescript
// Step 2 — create provider config
useCreateLlmProvider()
// mutation: POST /v1/settings/llm-providers { name, providerType, baseUrl?, apiKey, model, isDefault?, useCase? }
// returns: LlmConfig (includes .id for subsequent steps)

// Step 2 — test connection
useTestLlmProvider()
// mutation: POST /v1/settings/llm-providers/:id/test { id }
// returns: { success: boolean, latencyMs: number, error?: string }

// Step 3 — discover models
useDiscoverLlmModels()
// mutation: POST /v1/settings/llm-providers/:id/discover { id }
// returns: { models: DiscoveredModel[] }
// DiscoveredModel: { id: string, displayName?: string, capability: string[], contextWindow?: number }

// Step 4 — update provider with selected model
useUpdateLlmProvider()
// mutation: PUT /v1/settings/llm-providers/:id { model?, name?, isDefault? }

// Step 5 — benchmark
useBenchmarkLlmProvider()
// mutation: POST /v1/settings/llm-providers/:id/benchmark { id, data: { model?: string } }
// returns: { ttftMs: number, totalLatencyMs: number, qualityScore: number, model: string }
// qualityScore: 0-3 (0=no JSON, 1=partial JSON, 2=all keys, 3=all keys + valid enums)

// Step 6 — get routing (entries + suggestions)
useGetLlmRouting(tenantId, { query: { queryKey: [...] } })
// returns: { entries: RoutingEntry[], suggestions: Record<string, string | null> }
// RoutingEntry: { taskType, configId, modelOverride, effectiveModel, providerName }

// Step 6 — update routing
useUpdateLlmRouting()
// mutation: PUT /v1/settings/llm-routing { assignments: Array<{ taskType, configId, modelOverride? }> }
```

   - `<Input>` for provider nickname (wizardData.providerName)
