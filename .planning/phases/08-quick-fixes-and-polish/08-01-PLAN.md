  ```typescript
  // Extract domain from email: admin@acme.com → acme.com → acme (part before first dot)
  const emailDomain = email.split('@')[1];
  if (!emailDomain) {
    badRequest(res, "Invalid email address");
    return;
  }
  const domainSlug = emailDomain.split('.')[0]; // acme.com → acme

  const tenants = await db
    .select()
    .from(tenantsTable)
    .where(eq(tenantsTable.slug, domainSlug))
    .limit(1);

  const tenant = tenants[0];
  if (!tenant) {
    unauthorized(res, "No organization found for this email domain. Contact your administrator.");
    return;
  }
  ```
  The rest of the user lookup and JWT generation stays unchanged.

```typescript
// Replace the current select() with a join
const result = await db
```
