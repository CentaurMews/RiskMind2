<!-- API endpoints from Plan 02 -->

POST /v1/frameworks — Create framework
  Request: { name: string, version?: string, type?: string, description?: string }
  Response: 201 { id, name, version, type, description, ... }

POST /v1/frameworks/:id/import/preview — Parse file, return diff (no DB write)
  Request: multipart/form-data with field "file", query param "format" = "csv" | "json"
  Response: 200 { diff: { new: RawRequirement[], modified: { incoming, existing }[], unchanged: RawRequirement[] }, warnings: string[], totalIncoming: number }

POST /v1/frameworks/:id/import/apply — Apply import to DB
  Request: multipart/form-data with field "file", query param "format" = "csv" | "json"
  Response: 200 { imported: { new: number, modified: number, unchanged: number }, warnings: string[] }

PUT /v1/frameworks/:id/threshold — Update compliance threshold
  Request: { threshold: number } (0-100)
  Response: 200 { ...framework }

GET /v1/frameworks/:frameworkId/compliance-score — Now includes status field
  Response: { score, coverageScore, effectivenessScore, ..., status: "COMPLIANT" | "AT-RISK" | "NON-COMPLIANT" | null }

GET /v1/frameworks — List endpoint (already enriched in prior phase)
  Response: Array of { id, name, version, type, description, compliancePercentage, complianceThreshold, ... }
  NOTE: The list endpoint returns compliancePercentage and complianceThreshold per framework.
  Status can be derived client-side from these two fields using the same logic as getComplianceStatus:
    score >= threshold -> "COMPLIANT"
    score >= threshold - 15 -> "AT-RISK"
    score < threshold - 15 -> "NON-COMPLIANT"
    threshold is null -> no status label

   ```typescript
   function deriveComplianceStatus(score: number | null, threshold: number | null): "COMPLIANT" | "AT-RISK" | "NON-COMPLIANT" | null {
     if (score == null || threshold == null) return null;
     if (score >= threshold) return "COMPLIANT";
     if (score >= threshold - 15) return "AT-RISK";
     return "NON-COMPLIANT";
   }
   ```
   This avoids N+1 API calls (one per card) since the list endpoint already provides both values. Display the status as a colored badge on each FrameworkCard:
   - "COMPLIANT" -> green badge
   - "AT-RISK" -> amber badge
   - "NON-COMPLIANT" -> red badge
   - null (no threshold) -> no badge shown
