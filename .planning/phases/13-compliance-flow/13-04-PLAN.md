```typescript
const handleExportPdf = async () => {
  setExporting(true);
  try {
    // Dynamic import to avoid loading PDF renderer on initial page load (per research pitfall 4)
    const { pdf } = await import("@react-pdf/renderer");
    const { default: CompliancePdfReport } = await import("@/components/compliance/compliance-pdf-report");

    const blob = await pdf(
      <CompliancePdfReport
        frameworkName={framework.name}
        frameworkVersion={framework.version}
        score={scoreData}
        gaps={gapData}
        generatedAt={new Date().toISOString()}
      />
    ).toBlob();

    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = `${framework.name}-compliance-report.pdf`;
    a.click();
    URL.revokeObjectURL(url);
  } catch (err) {
    toast.error("Failed to generate PDF");
    console.error(err);
  } finally {
    setExporting(false);
  }
};
```

Add `exporting` state, show spinner on button while generating.
