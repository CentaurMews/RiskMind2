    ```typescript
    export interface BranchCondition {
      questionId: string;
      operator: "equals" | "contains" | "greater_than";
      value: string | number;
      action: "show" | "hide";
    }

    export interface QuestionOption {
      id: string;
      label: string;
      value: string;
      numericValue?: number; // normalized 0-1 for scoring
    }

    export interface AssessmentQuestion {
      id: string;
      sectionId: string;
      text: string;
      type: "text" | "boolean" | "multiple_choice" | "numeric";
      weight: number; // 0-10, default 5
      required: boolean;
      options?: QuestionOption[];
      numericMin?: number;
      numericMax?: number;
      conditions: BranchCondition[];
      isAiGenerated?: boolean;
      triggeredByQuestionId?: string;
    }

    export interface AssessmentSection {
      id: string;
      name: string;
      order: number;
      questions: AssessmentQuestion[];
    }

    export interface AssessmentTemplateQuestions {
      sections: AssessmentSection[];
      version: number;
    }

    export interface QuestionResponse {
      questionId: string;
      answer: string | boolean | number | string[];
      answeredAt: string; // ISO timestamp
    }

    export interface AiFollowUpRecord {
      id: string;
      triggeredByQuestionId: string;
      sectionId: string;
      question: AssessmentQuestion;
      generatedAt: string;
    }

    export interface AssessmentResponses {
      currentSectionIndex: number;
      responses: Record<string, QuestionResponse>;
      aiFollowUps: AiFollowUpRecord[];
      completedSections: string[];
    }

    export interface QuestionScore {
      questionId: string;
      score: number;
      weight: number;
    }

    export interface SectionScore {
      sectionId: string;
      name: string;
      score: number; // 0-100, rounded to 2 decimals
      questionScores: QuestionScore[];
    }

    export interface AssessmentScore {
      overall: number; // 0-100, rounded to 2 decimals
      sections: SectionScore[];
    }
    ```

    **Functions to export:**

    1. `isQuestionVisible(question, allResponses, allSectionQuestions)` — evaluates branching conditions:
       - No conditions → true
       - action="show": all conditions must pass for question to be visible (AND logic)
       - action="hide": if any condition passes, question is hidden
       - Operators: "equals" (String comparison), "contains" (substring), "greater_than" (numeric comparison)
       - If referenced question has no response: show-action → false (hidden until dependency answered), hide-action → true (visible until dependency answered)

    2. `normalizeAnswer(question, answer)` — returns 0.0-1.0:
       - boolean: true=1.0, false=0.0, null/undefined=0.0
       - numeric: (answer - numericMin) / (numericMax - numericMin), clamped 0-1. If no min/max, assume 0-10.
       - multiple_choice: find matching option's numericValue. If no numericValue, return (selectedIndex / totalOptions).
       - text: always 1.0 (qualitative, no numeric score)
       - null/undefined answer: 0.0

    3. `computeScore(template, responses)` — returns AssessmentScore:
       - For each section: filter to visible questions (isQuestionVisible), compute each question score as normalizeAnswer × weight, sum weighted scores / sum weights × 100, round to 2 decimals
       - Overall: average of section scores, round to 2 decimals
       - Empty sections (0 visible questions) excluded from overall average
