<!-- API endpoints from Plan 02 -->
GET /v1/assessments — list assessments (query: status, contextType)
GET /v1/assessments/:id — single assessment with template data
POST /v1/assessments — body: { templateId, contextType, contextId }

GET /v1/assessments/:id/results — returns { assessment, score: AssessmentScore, template }
